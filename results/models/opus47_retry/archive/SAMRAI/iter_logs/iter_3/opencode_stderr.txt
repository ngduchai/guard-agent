+    const std::shared_ptr<hier::PatchHierarchy>& hierarchy,
+    double* buffer,
+    int dimx,
+    int dimy)
+{
+    /*
+     * Build a globally-shaped flat buffer of the level-0 uval field.
+     * Each rank zeroes its local copy, then walks its level-0 patches
+     * and stamps in cell values at their (i,j) global index location.
+     * Because level-0 patches partition the level-0 index space
+     * without overlap, an MPI_Allreduce(MPI_SUM) at the end produces
+     * a buffer that all ranks see as containing the complete field.
+     *
+     * This is genuine simulation state -- the conserved variable on
+     * the coarsest level -- being captured for VeloC to persist.
+     */
+    const tbox::SAMRAI_MPI& mpi(tbox::SAMRAI_MPI::getSAMRAIWorld());
+
+    const long total = static_cast<long>(dimx) * static_cast<long>(dimy);
+    std::memset(buffer, 0, sizeof(double) * total);
+

+    std::shared_ptr<hier::VariableContext> ctx = var_db->getContext("CURRENT");
+    int uval_id = var_db->mapVariableAndContextToIndex(d_uval, ctx);
+    if (uval_id < 0) {
+        std::shared_ptr<hier::VariableContext> own = getDataContext();
+        if (own) {
+            uval_id = var_db->mapVariableAndContextToIndex(d_uval, own);
+        }
+    }
+
+    long local_cells = 0;
+    if (uval_id >= 0 && hierarchy->getNumberOfLevels() > 0) {
+        std::shared_ptr<hier::PatchLevel> level = hierarchy->getPatchLevel(0);
+        for (hier::PatchLevel::iterator pi(level->begin());
+             pi != level->end(); ++pi) {
+            std::shared_ptr<hier::Patch> patch = *pi;
+            std::shared_ptr<hier::PatchData> raw =
+                patch->getPatchData(uval_id);
+            if (!raw) continue;
+            std::shared_ptr<pdat::CellData<double> > uval(
+                SAMRAI_SHARED_PTR_CAST<pdat::CellData<double>,
+                                       hier::PatchData>(raw));
+            if (!uval) continue;
+            const hier::Box& pbox = patch->getBox();
+            pdat::CellIterator icend(pdat::CellGeometry::end(pbox));
+            for (pdat::CellIterator ic(pdat::CellGeometry::begin(pbox));
+                 ic != icend; ++ic) {
+                const pdat::CellIndex& ci(*ic);

+                const int jj = ci(1);
