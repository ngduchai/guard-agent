+    const std::shared_ptr<hier::PatchHierarchy>& hierarchy,
+    int* out_levels,
+    int* out_i,
+    int* out_j,
+    double* out_values,
+    long max_records)
+{
+    /*
+     * Walk every patch on every finer level (level>0) on this rank and
+     * append a (level, i_global, j_global, value) record to the parallel
+     * arrays for each interior cell.  This captures the genuine
+     * fine-level state of the simulation that VeloC then persists, so
+     * that the recovery path can reconstruct the same fine-resolution
+     * field around the moving sphere instead of having to derive it
+     * from the coarse level alone.
+     */
+    if (max_records <= 0) {
+        return 0;
+    }
+
+    hier::VariableDatabase* var_db = hier::VariableDatabase::getDatabase();
+    std::shared_ptr<hier::VariableContext> ctx = var_db->getContext("CURRENT");
+    int uval_id = var_db->mapVariableAndContextToIndex(d_uval, ctx);
+    if (uval_id < 0) {
+        std::shared_ptr<hier::VariableContext> own = getDataContext();
+        if (own) {
+            uval_id = var_db->mapVariableAndContextToIndex(d_uval, own);
+        }
+    }
+    if (uval_id < 0) {
+        return 0;
+    }
+
+    long count = 0;
+    const int num_levels = hierarchy->getNumberOfLevels();
+    for (int ln = 1; ln < num_levels; ++ln) {
+        std::shared_ptr<hier::PatchLevel> level = hierarchy->getPatchLevel(ln);
+        if (!level) continue;
+        for (hier::PatchLevel::iterator pi(level->begin());
+             pi != level->end(); ++pi) {
+            std::shared_ptr<hier::Patch> patch = *pi;
+            std::shared_ptr<hier::PatchData> raw =
+                patch->getPatchData(uval_id);
+            if (!raw) continue;
+            std::shared_ptr<pdat::CellData<double> > uval(
+                SAMRAI_SHARED_PTR_CAST<pdat::CellData<double>,
+                                       hier::PatchData>(raw));
+            if (!uval) continue;
+            const hier::Box& pbox = patch->getBox();
+            pdat::CellIterator icend(pdat::CellGeometry::end(pbox));
+            for (pdat::CellIterator ic(pdat::CellGeometry::begin(pbox));
+                 ic != icend; ++ic) {
+                if (count >= max_records) {
+                    return count;
+                }
+                const pdat::CellIndex& ci(*ic);
+                out_levels[count] = ln;
+                out_i[count]      = ci(0);
+                out_j[count]      = ci(1);
+                out_values[count] = (*uval)(ci);
+                ++count;
+            }
+        }
+    }
+    return count;
+}

+long
+LinAdv::restoreFineLevelsFromSparseRecords(
+    const std::shared_ptr<hier::PatchHierarchy>& hierarchy,
+    const int* in_levels,
+    const int* in_i,
+    const int* in_j,
+    const double* in_values,
+    long num_records)
+{
+    /*
+     * For each record in the (already cross-rank gathered) input
+     * arrays, find the patch on this rank's matching level whose box
+     * contains (i_global, j_global) and overwrite that cell's uval
+     * value.  Cells not owned by this rank silently fall through
+     * (their record will be applied on whichever rank does own the
+     * containing patch post-restart).
+     */
+    if (num_records <= 0) {
+        return 0;
+    }
+
+    hier::VariableDatabase* var_db = hier::VariableDatabase::getDatabase();
+    std::shared_ptr<hier::VariableContext> ctx = var_db->getContext("CURRENT");
+    int uval_id = var_db->mapVariableAndContextToIndex(d_uval, ctx);
+    if (uval_id < 0) {
+        std::shared_ptr<hier::VariableContext> own = getDataContext();
+        if (own) {
+            uval_id = var_db->mapVariableAndContextToIndex(d_uval, own);
+        }
+    }
+    if (uval_id < 0) {
+        return 0;
+    }
+
+    const int num_levels = hierarchy->getNumberOfLevels();
+
+    /*
+     * To avoid an O(N_records * N_patches) scan, pre-build a per-level
+     * list of (patch_pointer, box) pairs for levels we actually have.
+     * Then per record we do a linear scan over this rank's patches on
+     * that record's level, which is bounded by the per-rank patch count
+     * (a few dozen at most for this LinAdv configuration).
+     */
+    std::vector<std::vector<std::shared_ptr<pdat::CellData<double> > > >
+        cell_data_per_level(num_levels);
+    std::vector<std::vector<hier::Box> > boxes_per_level(num_levels);
+    for (int ln = 1; ln < num_levels; ++ln) {
+        std::shared_ptr<hier::PatchLevel> level = hierarchy->getPatchLevel(ln);
+        if (!level) continue;
+        for (hier::PatchLevel::iterator pi(level->begin());
+             pi != level->end(); ++pi) {
+            std::shared_ptr<hier::Patch> patch = *pi;
+            std::shared_ptr<hier::PatchData> raw =
+                patch->getPatchData(uval_id);
+            if (!raw) continue;
+            std::shared_ptr<pdat::CellData<double> > uval(
+                SAMRAI_SHARED_PTR_CAST<pdat::CellData<double>,
+                                       hier::PatchData>(raw));
+            if (!uval) continue;
+            cell_data_per_level[ln].push_back(uval);
+            boxes_per_level[ln].push_back(patch->getBox());
+        }
+    }
+
+    long touched = 0;
+    for (long r = 0; r < num_records; ++r) {
+        const int ln = in_levels[r];
+        if (ln < 1 || ln >= num_levels) continue;
+        const int ii = in_i[r];
+        const int jj = in_j[r];
+        const double v = in_values[r];
+        const size_t nbox = boxes_per_level[ln].size();
+        for (size_t b = 0; b < nbox; ++b) {
+            const hier::Box& pbox = boxes_per_level[ln][b];
+            if (ii < pbox.lower(0) || ii > pbox.upper(0) ||
+                jj < pbox.lower(1) || jj > pbox.upper(1)) {
+                continue;
+            }
+            pdat::CellIndex ci(pbox.lower());
+            ci(0) = ii;
+            ci(1) = jj;
+            (*cell_data_per_level[ln][b])(ci) = v;
+            ++touched;
+            break;  /* a cell can belong to at most one interior patch */
+        }
+    }
+
+    return touched;
+}
