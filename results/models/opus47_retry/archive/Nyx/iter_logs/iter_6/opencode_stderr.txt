+// Sidecar checkpoint: an atomic per-rank file in the VeloC persistent
+// directory holding the same blob as the VeloC checkpoint.
+//
+// WHY a sidecar?  Empirically, VeloC's `VELOC_Restart_test` returns a
+// "latest" version that, when fed to Recover_mem, can yield garbage
+// bytes (visible as a magic mismatch in earlier iterations) because the
+// kill landed mid-write and VeloC indexed an incomplete persistent
+// file.  Walking back through versions inside the VeloC API doesn't
+// always recover because every recent version may be touched by the
+// in-flight async write.  By writing our own POSIX file with the
+// fsync+rename atomic-replace pattern we get strict crash-atomicity:
+// either the new sidecar is fully present after rename, or the prior
+// complete sidecar remains untouched.  Either way, on restart we
+// always see a usable, complete checkpoint blob.
+//
+// Layout on disk (one file per rank in the VeloC persistent dir):
+//   <persistent>/nyx_sidecar_r<rank>.ckpt
+//
+// File contents are simply the same byte blob we'd hand to VeloC:
+//   [NyxVelocHeader][per-level state][int32 trailer magic]
+// We add a 12-byte file header with magic + version + payload size
+// so torn writes are detected even if the rename itself raced.

+static constexpr unsigned int NYX_SIDECAR_FILE_MAGIC = 0x4E594F31u; // 'NYO1'

+struct NyxSidecarFileHdr {
+    unsigned int file_magic;     // NYX_SIDECAR_FILE_MAGIC
+    unsigned int file_version;   // monotonic checkpoint counter
+    long         payload_bytes;  // bytes of blob payload after this hdr
+};

+{

+    std::snprintf(buf, sizeof(buf), "/nyx_sidecar_r%04d.ckpt", rank);

+// Write `blob` to the sidecar file atomically (write-then-rename).
+// Returns true on success.  Best-effort: failure does not propagate.
+static bool nyx_sidecar_write(const std::string& dir,
+                              int rank,
+                              unsigned int version_counter,
+                              const char* blob,
+                              long blob_bytes)
+{
+    if (dir.empty() || blob_bytes <= 0 || blob == nullptr) return false;
+    const std::string final_path = nyx_sidecar_path(dir, rank);
+    const std::string tmp_path   = final_path + ".tmp";
+
+    int fd = ::open(tmp_path.c_str(),
+                    O_WRONLY | O_CREAT | O_TRUNC,
+                    0644);
+    if (fd < 0) return false;
+
+    NyxSidecarFileHdr h;
+    h.file_magic     = NYX_SIDECAR_FILE_MAGIC;
+    h.file_version   = version_counter;
+    h.payload_bytes  = blob_bytes;
+    if (::write(fd, &h, sizeof(h)) != (ssize_t)sizeof(h)) {
+        ::close(fd); ::unlink(tmp_path.c_str()); return false;
+    }
+
+    // Write payload in chunks (in case of huge blobs).
+    long written = 0;
+    while (written < blob_bytes) {
+        const long chunk = std::min<long>(1L << 20, blob_bytes - written);
+        ssize_t n = ::write(fd, blob + written, (size_t)chunk);
+        if (n <= 0) {
+            ::close(fd); ::unlink(tmp_path.c_str()); return false;
+        }
+        written += n;
+    }
+    // fsync the data before rename so the new file content is durable.
+    ::fsync(fd);
+    ::close(fd);
+
+    // Atomic replace.
+    if (::rename(tmp_path.c_str(), final_path.c_str()) != 0) {
+        ::unlink(tmp_path.c_str());
+        return false;
+    }
+    return true;
+}
