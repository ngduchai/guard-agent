+// Write the raw checkpoint blob to the shadow file for this
+// rank+version.  We fdatasync before close so the bytes are
+// durable on /tmp before we return — matching the durability
+// expectation users have of a successful checkpoint.
+static bool nyx_shadow_write(const std::string& dir,
+                             int rank, int version,
+                             const char* data, size_t nbytes)
+{
+    if (dir.empty()) return false;
+    // mkdir is harmless if it already exists.
+    ::mkdir(dir.c_str(), 0755);
+    const std::string path = nyx_shadow_path(dir, rank, version);
+    // Write to .tmp then rename for atomic visibility.  This way
+    // a partial shadow file from a kill-mid-write never appears
+    // under its final name; the recovery scanner only sees fully
+    // written files.
+    const std::string tmp = path + ".tmp";
+    int fd = ::open(tmp.c_str(),
+                    O_WRONLY | O_CREAT | O_TRUNC, 0644);
+    if (fd < 0) return false;
+    size_t written = 0;
+    while (written < nbytes) {
+        ssize_t n = ::write(fd, data + written, nbytes - written);
+        if (n <= 0) {
+            ::close(fd);
+            ::unlink(tmp.c_str());
+            return false;
+        }
+        written += static_cast<size_t>(n);
+    }
+    ::fdatasync(fd);
+    ::close(fd);
+    if (::rename(tmp.c_str(), path.c_str()) != 0) {

+    }
