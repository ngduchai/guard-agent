+// Scan the persistent directory for the latest version number
+// encoded in any VeloC checkpoint file matching "<name>-*-v<N>.*"
+// pattern.  Used as a fallback when VELOC_Restart_test returns
+// <=0 despite the persistent dir containing valid checkpoints.
+// Returns 0 if no version found.
+static int nyx_scan_latest_version(const std::string& dir,
+                                   const std::string& name)
+{
+    if (dir.empty()) return 0;
+    DIR* d = opendir(dir.c_str());
+    if (!d) return 0;
+    int latest = 0;
+    const std::string prefix = name + "-";
+    struct dirent* ent;
+    while ((ent = readdir(d)) != nullptr) {
+        const char* nm = ent->d_name;
+        if (std::strncmp(nm, prefix.c_str(), prefix.size()) != 0) continue;
+        // Find "-v<digits>" anywhere after the prefix.
+        const char* v = std::strstr(nm, "-v");
+        if (!v) continue;
+        v += 2;
+        if (*v < '0' || *v > '9') continue;
+        int version = 0;
+        while (*v >= '0' && *v <= '9') {
+            version = version * 10 + (*v - '0');
+            ++v;
+        }
+        if (version > latest) latest = version;
+    }
+    closedir(d);
+    return latest;
+}
