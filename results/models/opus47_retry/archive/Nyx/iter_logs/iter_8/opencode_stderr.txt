+// Scan `dir` for the highest version number V such that a file
+// matching the pattern "<name>-<rank>-v<V>.*" exists for the
+// given rank.  Used by the recovery fallback to find a version
+// that THIS rank can actually restore from.  When combined with
+// ReduceIntMin across ranks, this picks the highest version that
+// EVERY rank can restore from.  Returns 0 if none found.
+//
+// VeloC checkpoint files follow the convention
+//   "<name>-<rank>-v<version>.<ext>"  (e.g. nyx-0-v23.dat).
+// We anchor on the "<name>-<rank>-v" prefix to avoid matching the
+// per-rank index of a different rank.
+static int nyx_scan_latest_rank_version(const std::string& dir,
+                                        const std::string& name,
+                                        int rank)
+{
+    if (dir.empty()) return 0;
+    DIR* d = opendir(dir.c_str());
+    if (!d) return 0;
+    int latest = 0;
+    char rank_prefix[128];
+    std::snprintf(rank_prefix, sizeof(rank_prefix),
+                  "%s-%d-v", name.c_str(), rank);
+    const size_t rplen = std::strlen(rank_prefix);
+    struct dirent* ent;
+    while ((ent = readdir(d)) != nullptr) {
+        const char* nm = ent->d_name;
+        if (std::strncmp(nm, rank_prefix, rplen) != 0) continue;
+        const char* v = nm + rplen;
+        if (*v < '0' || *v > '9') continue;
+        int version = 0;
+        while (*v >= '0' && *v <= '9') {
+            version = version * 10 + (*v - '0');
+            ++v;
+        }
+        if (version > latest) latest = version;
+    }
+    closedir(d);
+    return latest;
