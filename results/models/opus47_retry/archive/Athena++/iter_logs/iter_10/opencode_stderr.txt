+// mom3].  Caller MPI_Reduces across MPI_COMM_WORLD to get the global
+// sums on rank 0.
+static void athena_compute_history_locals(Mesh *pmesh, Real hst_local[4]) {
+  hst_local[0] = 0.0;
+  hst_local[1] = 0.0;
+  hst_local[2] = 0.0;

+    MeshBlock *pmb = pmesh->my_blocks(b);
+    AthenaArray<Real> &u = pmb->phydro->u;
+    AthenaArray<Real> vol(pmb->ncells1);
+    for (int k = pmb->ks; k <= pmb->ke; ++k) {
+      for (int j = pmb->js; j <= pmb->je; ++j) {
+        pmb->pcoord->CellVolume(k, j, pmb->is, pmb->ie, vol);
+        for (int i = pmb->is; i <= pmb->ie; ++i) {
+          const Real u_d  = u(IDN, k, j, i);

+          const Real u_my = u(IM2, k, j, i);
+          const Real u_mz = u(IM3, k, j, i);
+          hst_local[0] += vol(i) * u_d;
+          hst_local[1] += vol(i) * u_mx;
