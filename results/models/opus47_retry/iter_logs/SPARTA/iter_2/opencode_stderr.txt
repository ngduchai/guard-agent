+/* Veloc checkpoint disk-management helpers.
+ *
+ * The validator imposes a 10 GB cap on the combined size of the scratch +
+ * persistent VeloC directories.  SPARTA's checkpoint payload (the per-rank
+ * particle array) can easily be hundreds of MB; left unchecked, VeloC will
+ * accumulate every version forever and blow through the cap.  After we
+ * write a new checkpoint version we therefore scan the two configured
+ * directories and remove every "<name>-*-<old_version>.dat" file that
+ * predates the current version.  Only rank 0 actually unlinks - all ranks
+ * write into the same shared scratch/persistent paths so a single janitor
+ * is enough.
+ */
+static void sparta_veloc_parse_cfg_dirs(const char *cfg_path,
+                                        char *scratch_out, size_t scratch_sz,
+                                        char *persist_out, size_t persist_sz)
+{
+  scratch_out[0] = '\0';
+  persist_out[0] = '\0';
+  FILE *f = fopen(cfg_path, "r");
+  if (!f) return;
+  char line[1024];
+  while (fgets(line, sizeof(line), f)) {
+    char key[64], val[1024];
+    if (sscanf(line, " %63[^= \t] = %1023[^\n\r]", key, val) == 2) {
+      // strip leading/trailing whitespace from val
+      char *p = val; while (*p == ' ' || *p == '\t') p++;
+      char *e = p + strlen(p);
+      while (e > p && (e[-1] == ' ' || e[-1] == '\t' ||
+                       e[-1] == '\r' || e[-1] == '\n')) { *--e = '\0'; }
+      if (strcmp(key, "scratch") == 0)
+        snprintf(scratch_out, scratch_sz, "%s", p);
+      else if (strcmp(key, "persistent") == 0)
+        snprintf(persist_out, persist_sz, "%s", p);
+    }
+  }
+  fclose(f);
+}

+static void sparta_veloc_purge_dir(const char *dir, const char *name,
+                                   int keep_version)
+{
+  if (!dir || !dir[0]) return;
+  DIR *d = opendir(dir);
+  if (!d) return;
+  struct dirent *de;
+  size_t name_len = strlen(name);
+  while ((de = readdir(d)) != NULL) {
+    const char *fn = de->d_name;
+    if (fn[0] == '.') continue;
+    // accept files starting with "<name>-" or "<name>." (covers both
+    // common veloc naming conventions; if neither matches we skip).
+    if (strncmp(fn, name, name_len) != 0) continue;
+    char sep = fn[name_len];
+    if (sep != '-' && sep != '.' && sep != '_') continue;
+    // extract trailing version: ...<sep>v<digits>.dat or ...-<rank>-<v>.dat
+    // The last numeric run before ".dat" (or before end) is the version.
+    const char *dot = strrchr(fn, '.');
+    const char *end = dot ? dot : (fn + strlen(fn));
+    const char *q = end;
+    while (q > fn && q[-1] >= '0' && q[-1] <= '9') q--;
+    if (q == end) continue;
+    int v = atoi(q);
+    if (v > 0 && v < keep_version) {
+      char path[2048];
+      snprintf(path, sizeof(path), "%s/%s", dir, fn);
+      unlink(path);
+    }
+  }
+  closedir(d);
+}
