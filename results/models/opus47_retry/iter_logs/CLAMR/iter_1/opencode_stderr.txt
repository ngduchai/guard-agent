+static int read_checkpoint(int version)
+{
+   if (!veloc_active) return -1;
+
+   int rc = VELOC_Restart_begin(VELOC_CKPT_NAME, version);
+   if (rc != VELOC_SUCCESS) {
+      fprintf(stderr, "[VELOC] Restart_begin failed (rc=%d)\n", rc);
+      return -2;
+   }

+   char fname[VELOC_MAX_NAME];
+   char base[64];

+      fprintf(stderr, "[VELOC] Route_file (restart) failed (rc=%d)\n", rc);
+      VELOC_Restart_end(0);
+      return -3;
+   }

+   FILE *fp = fopen(fname, "rb");
+   if (!fp) {

+      VELOC_Restart_end(0);
+      return -4;
+   }
+
+   uint32_t magic = 0, fmt = 0;
+   if (fread(&magic, sizeof(uint32_t), 1, fp) != 1 ||
+       fread(&fmt,   sizeof(uint32_t), 1, fp) != 1 ||
+       magic != VELOC_CKPT_MAGIC ||
+       fmt   != VELOC_CKPT_FORMAT_VERSION) {
+      fprintf(stderr, "[VELOC] Bad magic/version in %s (got %08x / %u)\n", fname, magic, fmt);
+      fclose(fp);
+      VELOC_Restart_end(0);
+      return -5;
+   }
+
+   int saved_ncycle, saved_next_gc, saved_numpe, saved_mype, saved_noffset, nsz, ndp;
+   double saved_simTime, saved_deltaT, saved_Hi, saved_upper;
+   size_t saved_ncells, saved_ncells_glob;
+
+   fread(&saved_ncycle,  sizeof(int), 1, fp);
+   fread(&saved_next_gc, sizeof(int), 1, fp);
+   fread(&saved_simTime, sizeof(double), 1, fp);
+   fread(&saved_deltaT,  sizeof(double), 1, fp);
+   fread(&saved_Hi,      sizeof(double), 1, fp);
+   fread(&saved_upper,   sizeof(double), 1, fp);
+   fread(&saved_numpe,   sizeof(int), 1, fp);
+   fread(&saved_mype,    sizeof(int), 1, fp);
+   fread(&saved_ncells,  sizeof(size_t), 1, fp);
+   fread(&saved_noffset, sizeof(int), 1, fp);
+   fread(&saved_ncells_glob, sizeof(size_t), 1, fp);
+
+   if (saved_numpe != mesh->numpe) {
+      fprintf(stderr, "[VELOC] numpe mismatch: saved=%d current=%d (rank %d)\n",
+              saved_numpe, mesh->numpe, mesh->mype);
+      fclose(fp);
+      VELOC_Restart_end(0);
+      return -6;
+   }
+   if (saved_mype != mesh->mype) {
+      fprintf(stderr, "[VELOC] mype mismatch: saved=%d current=%d\n",
+              saved_mype, mesh->mype);
+      fclose(fp);
+      VELOC_Restart_end(0);
+      return -7;
+   }
+
+   fread(&nsz, sizeof(int), 1, fp);
+   mesh->nsizes.assign(nsz, 0);
+   if (nsz > 0) fread(&mesh->nsizes[0], sizeof(int), nsz, fp);
+   fread(&ndp, sizeof(int), 1, fp);
+   mesh->ndispl.assign(ndp, 0);
+   if (ndp > 0) fread(&mesh->ndispl[0], sizeof(int), ndp, fp);
+

+   // then refresh the C-pointer aliases (i, j, level, H, U, V) since the
+   // MallocPlus realloc may have moved them.
+   mesh->ncells = saved_ncells;
+   mesh->ncells_global = saved_ncells_glob;
+   mesh->noffset = saved_noffset;
+   mesh->resize(saved_ncells);
+   mesh->memory_reset_ptrs();
+   state->resize(saved_ncells);
+   state->memory_reset_ptrs();
+
+   if (saved_ncells > 0) {
+      fread(mesh->i,     sizeof(int),     saved_ncells, fp);
+      fread(mesh->j,     sizeof(int),     saved_ncells, fp);
+      fread(mesh->level, sizeof(uchar_t), saved_ncells, fp);
+      fread(state->H,    sizeof(state_t), saved_ncells, fp);
+      fread(state->U,    sizeof(state_t), saved_ncells, fp);
+      fread(state->V,    sizeof(state_t), saved_ncells, fp);
+   }
+
+   fclose(fp);
+
+   // Restore global scalars.
+   ncycle              = saved_ncycle;
+   next_graphics_cycle = saved_next_gc;
+   simTime             = saved_simTime;
+   deltaT              = saved_deltaT;
+   H_sum_initial       = saved_Hi;
+   upper_mass_diff_percentage = saved_upper;

+   // restored i/j/level are sufficient for these because they are the AMR
+   // primary data; everything else is recomputable.
+   mesh->calc_celltype(saved_ncells);
+   mesh->calc_spatial_coordinates(0);
+   mesh->calc_neighbors_local();
+
+   rc = VELOC_Restart_end(1);
+   if (rc != VELOC_SUCCESS) {
+      fprintf(stderr, "[VELOC] Restart_end failed (rc=%d)\n", rc);
+      return -8;
+   }
+   return 0;
+}
