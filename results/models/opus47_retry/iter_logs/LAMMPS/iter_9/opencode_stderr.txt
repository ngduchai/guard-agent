+// Atomic write to <path>.tmp + rename(2).
+static bool atomic_write(const std::string &final_path,
+                         const char *data, size_t len)
+{
+  std::string tmp_path = final_path + ".tmp";
+  FILE *fp = std::fopen(tmp_path.c_str(), "wb");
+  if (!fp) return false;
+  if (len > 0) {
+    if (std::fwrite(data, 1, len, fp) != len) { std::fclose(fp); return false; }
+  }
+  std::fflush(fp);
+  ::fsync(::fileno(fp));
+  std::fclose(fp);
+  if (::rename(tmp_path.c_str(), final_path.c_str()) != 0) return false;
+  return true;
+}

+// Read whole file into vector<char>.
+static bool read_whole_file(const std::string &path, std::vector<char> &out)
+{
+  FILE *fp = std::fopen(path.c_str(), "rb");
