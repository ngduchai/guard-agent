+int recoverCheckpoint(SimFlat* sim, int* iStep)
+{
+   int version = VELOC_Restart_test("comd", sim->nSteps + 1);
+   if (version == VELOC_FAILURE)
+      return 0;
+
+   if (VELOC_Restart_begin("comd", version) != VELOC_SUCCESS ||
+       VELOC_Recover_mem() != VELOC_SUCCESS ||
+       VELOC_Restart_end(1) != VELOC_SUCCESS)
+   {
+      fprintf(stderr, "Failed to recover VeloC checkpoint version %d.\n", version);
+      MPI_Abort(MPI_COMM_WORLD, 2);

+   computeForce(sim);
