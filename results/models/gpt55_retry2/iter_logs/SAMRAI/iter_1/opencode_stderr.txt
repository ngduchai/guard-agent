+getMaxLevels(const std::shared_ptr<algs::TimeRefinementIntegrator>& time_integrator)
+{

+   const std::shared_ptr<hier::PatchHierarchy> hierarchy(
+      time_integrator->getPatchHierarchy());

+void
+captureIntegratorVectors(
+   const std::shared_ptr<algs::TimeRefinementIntegrator>& time_integrator,
+   std::vector<double>& double_state,
+   std::vector<int>& int_state)
+{
+   const int max_levels = getMaxLevels(time_integrator);

+   int_state.assign(static_cast<size_t>(2 * max_levels), 0);

+   const std::shared_ptr<hier::PatchHierarchy> hierarchy(
+      time_integrator->getPatchHierarchy());
+   const int finest = hierarchy->getFinestLevelNumber();
+   for (int ln = 0; ln < max_levels; ++ln) {

+         double_state[max_levels + ln] = time_integrator->getLevelDtMax(ln);

+         double_state[3 * max_levels + ln] = time_integrator->getLevelSimTime(ln);
+         double_state[4 * max_levels + ln] = time_integrator->getLevelSimTime(ln);
+         int_state[ln] = time_integrator->getLevelStep(ln);

+std::vector<LinAdvFieldRegion>
+collectUvalRegions(
+   const std::shared_ptr<hier::PatchHierarchy>& hierarchy,
+   int uval_id)
+{
+   std::vector<LinAdvFieldRegion> regions;
+   int region_number = 0;
+   const int finest = hierarchy->getFinestLevelNumber();
+   for (int ln = 0; ln <= finest; ++ln) {
+      const std::shared_ptr<hier::PatchLevel>& level = hierarchy->getPatchLevel(ln);
+      for (hier::PatchLevel::Iterator p(level->begin()); p != level->end(); ++p) {
+         const std::shared_ptr<hier::Patch>& patch = *p;
+         std::shared_ptr<hier::PatchData> raw = patch->getPatchData(uval_id);
+         std::shared_ptr<pdat::CellData<double> > uval(
+            SAMRAI_SHARED_PTR_CAST<pdat::CellData<double>, hier::PatchData>(raw));
+         if (uval) {
+            LinAdvFieldRegion region;
+            region.id = LINADV_VELOC_FIELD_ID_BASE + region_number;
+            region.data = uval->getPointer();
+            region.count = uval->getArrayData().getOffset() * uval->getDepth();
+            regions.push_back(region);
+            ++region_number;
+         }
+      }
+   }
+   return regions;

+void
+protectLinAdvCheckpoint(
+   LinAdvCheckpointMeta& meta,
+   std::vector<double>& double_state,
+   std::vector<int>& int_state,
+   std::vector<LinAdvFieldRegion>& regions)
+{
+   VELOC_Mem_clear();
+   VELOC_Mem_protect(LINADV_VELOC_META_ID, &meta, 1, sizeof(meta));
+   VELOC_Mem_protect(LINADV_VELOC_META_ID + 1,
+      double_state.data(), double_state.size(), sizeof(double));
+   VELOC_Mem_protect(LINADV_VELOC_META_ID + 2,
+      int_state.data(), int_state.size(), sizeof(int));
+   for (size_t i = 0; i < regions.size(); ++i) {
+      VELOC_Mem_protect(regions[i].id, regions[i].data, regions[i].count, sizeof(double));
+   }
+}

+void
+writeLinAdvCheckpoint(
+   int version,
+   double dt_now,
+   const std::shared_ptr<algs::TimeRefinementIntegrator>& time_integrator,
+   const std::shared_ptr<hier::PatchHierarchy>& hierarchy,
+   int uval_id)
+{
+   std::vector<double> double_state;
+   std::vector<int> int_state;
+   captureIntegratorVectors(time_integrator, double_state, int_state);
+   std::vector<LinAdvFieldRegion> regions = collectUvalRegions(hierarchy, uval_id);
+
+   LinAdvCheckpointMeta meta;
+   meta.dt_now = dt_now;
+   meta.integrator_time = time_integrator->getIntegratorTime();
+   meta.integrator_step = time_integrator->getIntegratorStep();
+   meta.patch_count = static_cast<int>(regions.size());
+   meta.value_count = 0;
+   for (size_t i = 0; i < regions.size(); ++i) {
+      meta.value_count += static_cast<int>(regions[i].count);
+   }
+   meta.vector_count = getMaxLevels(time_integrator);
+
+   protectLinAdvCheckpoint(meta, double_state, int_state, regions);
+   if (VELOC_Checkpoint(LINADV_VELOC_NAME, version) != VELOC_SUCCESS) {
+      TBOX_ERROR("VeloC checkpoint failed at LinAdv version " << version << std::endl);
+   }
+}

+bool
+recoverLinAdvCheckpoint(
+   int version,

+   const std::shared_ptr<algs::TimeRefinementIntegrator>& time_integrator,
+   const std::shared_ptr<hier::PatchHierarchy>& hierarchy,
+   int uval_id)
+{
+   std::vector<double> double_state;
+   std::vector<int> int_state;
+   captureIntegratorVectors(time_integrator, double_state, int_state);
+   std::vector<LinAdvFieldRegion> regions = collectUvalRegions(hierarchy, uval_id);
+
+   LinAdvCheckpointMeta meta;
+   meta.dt_now = dt_now;
+   meta.integrator_time = time_integrator->getIntegratorTime();
+   meta.integrator_step = time_integrator->getIntegratorStep();
+   meta.patch_count = static_cast<int>(regions.size());
+   meta.value_count = 0;
+   meta.vector_count = getMaxLevels(time_integrator);
+
+   protectLinAdvCheckpoint(meta, double_state, int_state, regions);
+   if (VELOC_Restart(LINADV_VELOC_NAME, version) != VELOC_SUCCESS) {
+      return false;
+   }
+
+   if (meta.patch_count != static_cast<int>(regions.size()) ||
+       meta.vector_count != getMaxLevels(time_integrator)) {
+      TBOX_ERROR("VeloC restart metadata does not match the reconstructed LinAdv hierarchy" << std::endl);
+   }
+
+   dt_now = meta.dt_now;
+   restoreIntegratorVectors(time_integrator, double_state, int_state,
+      meta.integrator_time, meta.integrator_step);
+   return true;
+}
