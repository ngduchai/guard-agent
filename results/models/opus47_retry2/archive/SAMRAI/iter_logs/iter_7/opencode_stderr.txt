+// Sidecar bookkeeping file: a plain-text dump of the four ckpt
+// scalars (iteration_num, loop_time, dt_now, num_failures) written
+// per-rank.  Used as a backup if VELOC_Restart can't recover the

+// crash time).  Written ONLY after the per-rank physics file write
+// has been completed successfully.
+struct BookkeepingSidecar {
+    int    iteration_num;
+    double loop_time;
+    double dt_now;
+    int    num_failures;
+    int    version;
+};
+
+static std::string sidecarFilePath(int version, int rank)
+{
+    char path[1024];
+    std::snprintf(path, sizeof(path),
+                  "%s/linadv-bk-v%d-r%d.txt",
+                  kPersistentDir, version, rank);
+    return std::string(path);
+}
+
+static bool writeSidecar(int version, int rank,
+                         const BookkeepingSidecar& bk)
+{
+    std::string path = sidecarFilePath(version, rank);
+    std::string tmp = path + ".tmp";
+    std::FILE* fp = std::fopen(tmp.c_str(), "w");
+    if (!fp) return false;
+    std::fprintf(fp, "%d %.17g %.17g %d %d\n",
+                 bk.iteration_num, bk.loop_time, bk.dt_now,
+                 bk.num_failures, bk.version);
+    std::fflush(fp);
+    int fd = ::fileno(fp);
+    if (fd >= 0) {
+#ifdef _POSIX_SYNCHRONIZED_IO
+        ::fsync(fd);
+#endif
+    }
+    std::fclose(fp);
+    return std::rename(tmp.c_str(), path.c_str()) == 0;
+}
+
+static bool readSidecar(int version, int rank, BookkeepingSidecar& out)
+{
+    std::string path = sidecarFilePath(version, rank);
+    std::FILE* fp = std::fopen(path.c_str(), "r");
+    if (!fp) return false;
+    int n = std::fscanf(fp, "%d %lf %lf %d %d",
+                        &out.iteration_num, &out.loop_time,
+                        &out.dt_now, &out.num_failures,
+                        &out.version);
+    std::fclose(fp);
+    return n == 5;
+}
