+// Scan a VeloC checkpoint directory for files matching
+// "<name>-<rank>-<version>.<suffix>" and return the maximum version found
+// for this rank.  Returns -1 if none found.
+static int clamr_scan_dir_for_latest_version(const char *dirpath,
+                                             const char *name,
+                                             int rank) {
+   DIR *d = opendir(dirpath);
+   if (!d) {
+      dbg("scan: opendir(%s) failed errno=%d", dirpath, errno);
+      return -1;
+   }
+   int max_ver = -1;
+   char prefix[256];
+   int plen = snprintf(prefix, sizeof(prefix), "%s-%d-", name, rank);
+   struct dirent *ent;
+   while ((ent = readdir(d)) != NULL) {
+      if (strncmp(ent->d_name, prefix, plen) != 0) continue;
+      // Parse version portion: substring after prefix up to '.'
+      const char *vstart = ent->d_name + plen;
+      char *vend = NULL;
+      long v = strtol(vstart, &vend, 10);
+      if (vend == vstart) continue;
+      if (*vend != '.' && *vend != '\0') continue;
+      if (v <= 0) continue;
+      if ((int)v > max_ver) max_ver = (int)v;
+   }
+   closedir(d);
+   return max_ver;
