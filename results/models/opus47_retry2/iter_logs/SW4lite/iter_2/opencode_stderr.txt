+// Delete VeloC checkpoint files older than keep_version for the given name.
+// VeloC file names embed the version as the last token before the extension,
+// e.g. "<name>-<rank>-<version>.dat", "<name>-<rank>-<version>.meta".
+// We match files starting with "<name>-" and parse the integer that appears
+// immediately before the final '.': any file with version < keep_version is
+// unlinked.  Rank-0 alone scans the shared dirs.
+static void sw4_purge_old_veloc(const char *ckpt_name, int keep_version, int my_rank)
+{
+   if( my_rank != 0 ) return;
+   if( keep_version <= 0 ) return;
+   std::string scratch, persistent;
+   sw4_read_veloc_dirs(scratch, persistent);
+   const char *dirs[2] = { scratch.c_str(), persistent.c_str() };
+   const size_t nlen = std::strlen(ckpt_name);
+   for( int di = 0; di < 2; di++ )
+   {
+      const char *dpath = dirs[di];
+      if( !dpath || dpath[0] == 0 ) continue;
+      DIR *d = opendir(dpath);
+      if( !d ) continue;
+      struct dirent *ent;
+      while( (ent = readdir(d)) != NULL )
+      {
+         const char *fname = ent->d_name;
+         if( fname[0] == '.' ) continue;
+         // Must start with "<name>-".
+         if( std::strncmp(fname, ckpt_name, nlen) != 0 ) continue;
+         if( fname[nlen] != '-' ) continue;
+         // Find last '.' (extension) and parse integer immediately before it.
+         const char *dot = std::strrchr(fname, '.');
+         if( !dot || dot == fname ) continue;
+         // Walk backwards from dot to find start of integer token.
+         const char *p = dot;
+         while( p > fname && std::isdigit((unsigned char)*(p-1)) ) p--;
+         if( p == dot ) continue;          // no digits
+         if( p == fname || *(p-1) != '-' ) continue; // not preceded by '-'
+         int ver = std::atoi(p);
+         if( ver < keep_version )
+         {
+            std::string full = std::string(dpath) + "/" + fname;
+            ::unlink(full.c_str());
+         }
+      }
+      closedir(d);
