+// ============================================================================
+// VeloC checkpoint / restart support
+// ----------------------------------------------------------------------------
+// Strategy
+//   * We always run amrptr->init(...) so the box hierarchy and the
+//     StateDescriptor system come up exactly as in a fresh run.  This sets up
+//     identical BoxArrays / DistributionMaps as long as the inputs file is
+//     unchanged, so the in-memory layout of each MultiFab is byte-for-byte
+//     reproducible across the fail and the recovery process.
+//   * After init we allocate flat host buffers sized to hold the State_Type
+//     and DiagEOS_Type MultiFabs of every level (including ghosts -- we just
+//     copy the entire FArrayBox storage).  These buffers are registered with
+//     VeloC via VELOC_Mem_protect together with a small metadata buffer
+//     (cumtime, per-level dt and step count, finest_level).
+//   * On startup we call VELOC_Restart_test; if a checkpoint exists we
+//     VELOC_Restart it, copy the buffers back into the live MultiFabs, and
+//     restore cumtime / level_steps / dt_level on the Amr object.  Otherwise
+//     we keep the freshly-initialised state.
+//   * After every coarseTimeStep we re-serialise into the buffers and call
+//     VELOC_Checkpoint with the new coarse-step number as the version.  If
+//     the process is then killed, restart picks up the latest version and
+//     resumes from the saved physical time, reproducing the same final
+//     validation signature.

+// Compute the byte size of the flat buffer needed to hold every FArrayBox of
+// state_type across all levels of amrptr.  We iterate exactly the same way
+// the serialize/deserialize routines do, so the totals agree.
+static std::size_t computeStateBytes(amrex::Amr* amrptr, int state_type)
+{
+    using namespace amrex;
+    std::size_t total = 0;
+    const int nlev = amrptr->finestLevel() + 1;
+    for (int lev = 0; lev < nlev; ++lev) {
+        AmrLevel& al = amrptr->getLevel(lev);
+        MultiFab& mf = al.get_new_data(state_type);
+        for (MFIter mfi(mf, false); mfi.isValid(); ++mfi) {
+            const FArrayBox& fab = mf[mfi];
+            total += fab.nBytes();
+        }
+    }
+    return total;

+static void serializeState(amrex::Amr* amrptr, int state_type,
+                           char* buf, std::size_t bufsz)
+{
+    using namespace amrex;
+    std::size_t off = 0;
+    const int nlev = amrptr->finestLevel() + 1;
+    for (int lev = 0; lev < nlev; ++lev) {
+        AmrLevel& al = amrptr->getLevel(lev);
+        MultiFab& mf = al.get_new_data(state_type);

+                             const char* buf, std::size_t bufsz)
+{
+    using namespace amrex;

+    }
+    AMREX_ALWAYS_ASSERT(off == bufsz);

+    }
