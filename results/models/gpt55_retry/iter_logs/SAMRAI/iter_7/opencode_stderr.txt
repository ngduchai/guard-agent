+   const std::vector< std::vector<SavedPatchRecord> >& saved_patches,
+   const std::shared_ptr<hier::PatchHierarchy>& patch_hierarchy)
+{

+      patch_hierarchy->removePatchLevel(patch_hierarchy->getNumberOfLevels() - 1);
+   }
+
+   const tbox::Dimension& dim = patch_hierarchy->getDim();
+   std::vector<int> lower(dim.getValue());
+   std::vector<int> upper(dim.getValue());
+   for (int ln = 0; ln < static_cast<int>(saved_patches.size()); ++ln) {
+      hier::IntVector ratio = (ln == 0) ?
+         hier::IntVector(dim, 1) :
+         patch_hierarchy->getRatioToCoarserLevel(ln) *
+            patch_hierarchy->getPatchLevel(ln - 1)->getRatioToLevelZero();
+      std::shared_ptr<hier::BoxLevel> box_level(
+         new hier::BoxLevel(ratio,

+      for (std::vector<SavedPatchRecord>::const_iterator it = saved_patches[ln].begin();
+           it != saved_patches[ln].end(); ++it) {
+         if (it->owner != tbox::SAMRAI_MPI::getSAMRAIWorld().getRank()) {
+            continue;
+         }
+         for (unsigned int axis = 0; axis < dim.getValue(); ++axis) {
