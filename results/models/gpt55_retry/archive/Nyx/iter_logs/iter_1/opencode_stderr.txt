+static void nyxVelocFillMetadata (amrex::Amr* amrptr, NyxVelocMetadata& meta)
+{
+    std::memset(&meta, 0, sizeof(meta));
+    meta.finest_level = amrptr->finestLevel();
+    meta.cum_time = static_cast<double>(amrptr->cumTime());
+    meta.start_time = static_cast<double>(amrptr->startTime());
+    const int nlev = std::min(meta.finest_level + 1, 16);
+    for (int lev = 0; lev < nlev; ++lev) {
+        meta.level_steps[lev] = amrptr->levelSteps(lev);
+        meta.level_count[lev] = amrptr->levelCount(lev);
+        meta.dt_level[lev] = static_cast<double>(amrptr->dtLevel(lev));
+    }

+static std::vector<NyxVelocRegion> nyxVelocProtectState (amrex::Amr* amrptr,
+                                                          NyxVelocMetadata& meta)
+{
+    nyxVelocFillMetadata(amrptr, meta);
+    nyxVelocAbortIfFailed(VELOC_Mem_clear(), "VELOC_Mem_clear");
+    nyxVelocAbortIfFailed(VELOC_Mem_protect(NyxVelocMetadataId, &meta, 1, sizeof(meta)),
+                          "VELOC_Mem_protect(metadata)");
+
+    std::vector<NyxVelocRegion> regions;
+    int id = NyxVelocBaseFabId;
+    const int nlev = amrptr->finestLevel() + 1;
+    for (int lev = 0; lev < nlev; ++lev) {
+        amrex::MultiFab& mf = amrptr->getLevel(lev).get_new_data(State_Type);
+        for (amrex::MFIter mfi(mf); mfi.isValid(); ++mfi) {
+            amrex::FArrayBox& fab = mf[mfi];
+            regions.push_back({id, fab.dataPtr(), static_cast<std::size_t>(fab.size())});

+        }
