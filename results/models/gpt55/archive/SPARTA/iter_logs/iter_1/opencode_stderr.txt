+int Update::veloc_restart_if_available(bigint maxstep)
+{
+  if (maxstep <= 0) return 0;
+
+  int version = VELOC_Restart_test(SPARTA_VELOC_NAME,static_cast<int>(maxstep));
+  if (version < 0) return 0;
+
+  SpartaVelocMeta meta;
+  memset(&meta,0,sizeof(meta));
+
+  VELOC_Mem_clear();
+  VELOC_Mem_protect(SPARTA_VELOC_META_ID,&meta,1,sizeof(meta));
+
+  if (VELOC_Restart_begin(SPARTA_VELOC_NAME,version) != 0) return 0;
+
+  int ids[1] = {SPARTA_VELOC_META_ID};
+  int ok = VELOC_Recover_selective(VELOC_RECOVER_SOME,ids,1);
+  if (ok == 0) {
+    int bytes = VELOC_Recover_size(SPARTA_VELOC_PARTICLE_ID);
+    int needed = 0;
+    if (bytes > 0) needed = bytes / static_cast<int>(sizeof(Particle::OnePart));
+    if (needed > particle->maxlocal)
+      particle->grow(needed - particle->nlocal);
+
+    particle->nlocal = needed;
+    particle->nglobal = meta.nglobal;
+    ntimestep = meta.ntimestep;
+
+    VELOC_Mem_protect(SPARTA_VELOC_PARTICLE_ID,particle->particles,
+                      particle->nlocal,sizeof(Particle::OnePart));
+    ids[0] = SPARTA_VELOC_PARTICLE_ID;
+    ok = VELOC_Recover_selective(VELOC_RECOVER_SOME,ids,1);
+    if (ok == 0) {
+      particle->sorted = 0;
+      particle->sort();
+    }
+  }

+  VELOC_Restart_end(ok == 0);
+  VELOC_Mem_clear();
