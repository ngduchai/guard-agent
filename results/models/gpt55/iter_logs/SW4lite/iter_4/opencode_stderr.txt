+{

+      return VELOC_FAILURE;
 
+   FILE* f = std::fopen(routed, "wb");
+   if( f == NULL )
+      return VELOC_FAILURE;
+
+   const int magic = 0x53573443;
+   const int version = 1;
+   const int nGrids = checkpointU.size();
+   int ok = 1;
+   ok = ok && std::fwrite(&magic, sizeof(magic), 1, f) == 1;
+   ok = ok && std::fwrite(&version, sizeof(version), 1, f) == 1;
+   ok = ok && std::fwrite(&completedTimeStep, sizeof(completedTimeStep), 1, f) == 1;
+   ok = ok && std::fwrite(&t, sizeof(t), 1, f) == 1;

+   for( int g = 0 ; ok && g < nGrids ; g++ )

+      size_t umSize = checkpointUm[g].size();
