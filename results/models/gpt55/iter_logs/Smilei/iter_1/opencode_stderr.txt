+static void smileiSerializeState( const std::string &path, VectorPatch &vecPatches, SimWindow *simWindow,
+                                  unsigned int completed_step, double time_prim, double time_dual, int count_dlb )
+{
+    std::ofstream out( path.c_str(), std::ios::binary | std::ios::trunc );
+    if( !out ) {
+        throw std::runtime_error( "failed to open VeloC checkpoint payload for writing" );
+    }
+    const uint64_t magic = 0x534d494c45495643ULL;
+    const uint32_t version = 1;
+    smileiWriteScalar( out, magic );
+    smileiWriteScalar( out, version );
+    smileiWriteScalar( out, completed_step );
+    smileiWriteScalar( out, time_prim );
+    smileiWriteScalar( out, time_dual );
+    smileiWriteScalar( out, count_dlb );
+    const double x_moved = simWindow ? simWindow->getXmoved() : 0.0;
+    const unsigned int n_moved = simWindow ? simWindow->getNmoved() : 0;
+    smileiWriteScalar( out, x_moved );
+    smileiWriteScalar( out, n_moved );
+
+    const uint64_t npatches = vecPatches.size();
+    smileiWriteScalar( out, npatches );
+    for( unsigned int ipatch = 0; ipatch < vecPatches.size(); ++ipatch ) {
+        Patch *patch = vecPatches( ipatch );
+        smileiWriteScalar( out, patch->hindex );
+        const uint8_t has_rand = patch->rand_ != nullptr ? 1 : 0;
+        smileiWriteScalar( out, has_rand );
+        if( has_rand ) {
+            smileiWriteScalar( out, patch->rand_->xorshift32_state );
+        }
+        ElectroMagn *em = patch->EMfields;
+        const uint64_t nfields = em ? em->allFields.size() : 0;
+        smileiWriteScalar( out, nfields );
+        for( unsigned int ifield = 0; ifield < nfields; ++ifield ) {
+            smileiWriteField( out, em->allFields[ifield] );
+        }
+        const uint64_t nspecies = patch->vecSpecies.size();
+        smileiWriteScalar( out, nspecies );
+        for( unsigned int ispec = 0; ispec < patch->vecSpecies.size(); ++ispec ) {
+            smileiWriteParticles( out, patch->vecSpecies[ispec]->particles );
+        }
+    }

+static void smileiDeserializeState( const std::string &path, VectorPatch &vecPatches, SimWindow *simWindow,
+                                    unsigned int &completed_step, double &time_prim, double &time_dual, int &count_dlb )
+{
+    std::ifstream in( path.c_str(), std::ios::binary );
+    if( !in ) {
+        throw std::runtime_error( "failed to open VeloC checkpoint payload for reading" );
+    }
+    uint64_t magic = 0;
+    uint32_t version = 0;
+    smileiReadScalar( in, magic );
+    smileiReadScalar( in, version );
+    if( magic != 0x534d494c45495643ULL || version != 1 ) {
+        throw std::runtime_error( "VeloC checkpoint payload has an unsupported format" );
+    }
+    smileiReadScalar( in, completed_step );
+    smileiReadScalar( in, time_prim );
+    smileiReadScalar( in, time_dual );
+    smileiReadScalar( in, count_dlb );
+    double x_moved = 0.0;
+    unsigned int n_moved = 0;
+    smileiReadScalar( in, x_moved );
+    smileiReadScalar( in, n_moved );
+    if( simWindow ) {
+        simWindow->setXmoved( x_moved );
+        simWindow->setNmoved( static_cast<int>( n_moved ) );
+    }
+
+    uint64_t npatches = 0;
+    smileiReadScalar( in, npatches );
+    if( npatches != vecPatches.size() ) {
+        throw std::runtime_error( "VeloC checkpoint patch layout does not match current run" );
+    }
+    for( unsigned int ipatch = 0; ipatch < vecPatches.size(); ++ipatch ) {
+        Patch *patch = vecPatches( ipatch );
+        unsigned int hindex = 0;
+        smileiReadScalar( in, hindex );
+        if( hindex != patch->hindex ) {
+            throw std::runtime_error( "VeloC checkpoint patch hindex does not match current run" );
+        }
+        uint8_t has_rand = 0;
+        smileiReadScalar( in, has_rand );
+        if( has_rand ) {
+            uint32_t rand_state = 0;
+            smileiReadScalar( in, rand_state );
+            if( patch->rand_ ) {
+                patch->rand_->xorshift32_state = rand_state;
+            }
+        }
+        ElectroMagn *em = patch->EMfields;
+        uint64_t nfields = 0;
+        smileiReadScalar( in, nfields );
+        if( em == nullptr || nfields != em->allFields.size() ) {
+            throw std::runtime_error( "VeloC checkpoint electromagnetic-field layout mismatch" );
+        }
+        for( unsigned int ifield = 0; ifield < nfields; ++ifield ) {
+            smileiReadField( in, em->allFields[ifield] );
+        }
+        uint64_t nspecies = 0;
+        smileiReadScalar( in, nspecies );
+        if( nspecies != patch->vecSpecies.size() ) {
+            throw std::runtime_error( "VeloC checkpoint species layout mismatch" );
+        }
+        for( unsigned int ispec = 0; ispec < patch->vecSpecies.size(); ++ispec ) {
+            smileiReadParticles( in, patch->vecSpecies[ispec]->particles );
+        }
+    }
+}

+static bool smileiVelocCheckpoint( VectorPatch &vecPatches, SimWindow *simWindow, SmileiMPI &smpi,
+                                   unsigned int completed_step, double time_prim, double time_dual, int count_dlb )
+{
+    char routed[4096];
+    char original[128];
+    std::snprintf( original, sizeof( original ), "smilei_state_rank_%06d.bin", smpi.getRank() );
+    if( VELOC_Checkpoint_begin( smileiVelocName(), static_cast<int>( completed_step ) ) != 0 ) {
+        WARNING( "VeloC checkpoint begin failed at step " << completed_step );
+        return false;
+    }
+    if( VELOC_Route_file( original, routed ) != 0 ) {
+        WARNING( "VeloC route file failed at step " << completed_step );
+        VELOC_Checkpoint_end( 0 );
+        return false;
+    }
+    bool ok = true;
+    try {
+        smileiSerializeState( routed, vecPatches, simWindow, completed_step, time_prim, time_dual, count_dlb );
+    } catch( const std::exception &e ) {
+        WARNING( "VeloC checkpoint serialization failed at step " << completed_step << ": " << e.what() );
+        ok = false;
+    }
+    if( VELOC_Checkpoint_end( ok ? 1 : 0 ) != 0 ) {
+        WARNING( "VeloC checkpoint end failed at step " << completed_step );
+        ok = false;
+    }
+    return ok;
+}

+static bool smileiVelocRecover( VectorPatch &vecPatches, SimWindow *simWindow, SmileiMPI &smpi,

+        ok = false;
+    }
+    if( ok && smpi.isMaster() ) {
+        MESSAGE( "Restarted Smilei from VeloC checkpoint version " << version );
+    }
+    return ok;
