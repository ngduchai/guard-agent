void MeshBlockTree::ResetToRootLevel() {
  // This is called on the ROOT of the tree (level=0)
  // Traverse down to root_level, then delete all children of root_level nodes
  if (loc_.level == pmesh_->root_level) {
    if (pleaf_ != nullptr) {
      for (int n = 0; n < nleaf_; n++) delete pleaf_[n];
      delete [] pleaf_;
      pleaf_ = nullptr;
    }
    return;
  }
  if (pleaf_ != nullptr) {
    for (int n = 0; n < nleaf_; n++) {
      if (pleaf_[n] != nullptr)
        pleaf_[n]->ResetToRootLevel();
    }
  }
}
