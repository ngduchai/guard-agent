+//! \fn void MeshBlockTree::ResetToRootLevel()
+//! \brief delete all refined nodes, resetting the tree to only root-level leaf nodes.
+//!        Called on the logical root (level=0) to undo all AMR refinements.
+
+void MeshBlockTree::ResetToRootLevel() {
+  if (loc_.level == pmesh_->root_level) {
+    // This is a root-level node: delete all its refined children, if any
+    if (pleaf_ != nullptr) {
+      for (int n = 0; n < nleaf_; ++n)
+        delete pleaf_[n];
+      delete [] pleaf_;
+      pleaf_ = nullptr;
+    }
+    return;
+  }
+  // Not yet at root_level: recurse into existing children
+  if (pleaf_ != nullptr) {
+    for (int n = 0; n < nleaf_; ++n) {
+      if (pleaf_[n] != nullptr)
+        pleaf_[n]->ResetToRootLevel();
+    }
+  }
