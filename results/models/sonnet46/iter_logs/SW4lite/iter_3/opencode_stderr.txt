+// We scan the directory and delete any files matching the name prefix

+static void pruneVelocCheckpoints(const char* dir, const char* ckpt_name, int keep_version)
+{
+   DIR* dp = opendir(dir);
+   if (!dp) return;
+   struct dirent* de;
+   while ((de = readdir(dp)) != NULL)
+   {
+      // VeloC file pattern: <something>-<name>-<version>.veloc
+      // We look for files containing the checkpoint name and ending in .veloc
+      const char* fname = de->d_name;
+      // Check it ends with .veloc
+      size_t flen = strlen(fname);
+      if (flen < 6 || strcmp(fname + flen - 6, ".veloc") != 0)
+         continue;
+      // Find the checkpoint name in the filename
+      const char* namepos = strstr(fname, ckpt_name);
+      if (!namepos) continue;

+      const char* aftername = namepos + strlen(ckpt_name);
