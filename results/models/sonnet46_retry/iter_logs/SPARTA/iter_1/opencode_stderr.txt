+/* ----------------------------------------------------------------------
+   Write a VeloC checkpoint of ntimestep and all local particles.
+   Each rank writes its own file via VELOC_Route_file.
+   Returns 1 on success, 0 on failure.
+------------------------------------------------------------------------- */
+static int sparta_veloc_checkpoint(bigint ntimestep, int nlocal,
+                                   Particle::OnePart *particles)
+{
+  /* Determine checkpoint version from ntimestep (must be positive integer) */
+  int version = (int)(ntimestep & 0x7FFFFFFF);
+  if (version <= 0) version = 1;
+
+  if (VELOC_Checkpoint_begin("sparta_ckpt", version) != VELOC_SUCCESS)
+    return 0;
+
+  /* Get the routed file path for this rank */
+  char routed[VELOC_MAX_NAME];
+  if (VELOC_Route_file("sparta_state", routed) != VELOC_SUCCESS) {
+    VELOC_Checkpoint_end(0);
+    return 0;
+  }
+
+  /* Write binary state: ntimestep, nlocal, then all particle data */
+  FILE *fp = fopen(routed, "wb");
+  if (!fp) {
+    VELOC_Checkpoint_end(0);
+    return 0;
+  }
+  fwrite(&ntimestep, sizeof(bigint), 1, fp);
+  fwrite(&nlocal, sizeof(int), 1, fp);
+  if (nlocal > 0)
+    fwrite(particles, sizeof(Particle::OnePart), nlocal, fp);
+  fclose(fp);
+
+  if (VELOC_Checkpoint_end(1) != VELOC_SUCCESS)
+    return 0;
+  return 1;
