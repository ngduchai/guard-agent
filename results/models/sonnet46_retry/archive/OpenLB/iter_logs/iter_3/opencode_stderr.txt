+/// @brief Purge old checkpoint files from scratch and persistent dirs,
+/// keeping only the current version to bound disk usage.
+/// Called only by rank 0, then broadcast result to ensure all ranks wait.
+static void purge_old_checkpoints(const std::string& name, int keep_version)
+{
+  // Only purge files that belong to older versions of this checkpoint name.
+  // VeloC file naming: <name>-<version>-<rank>.ext in scratch/persistent.
+  // We delete any file that starts with the name prefix but does NOT contain
+  // the keep_version.
+
+  // Read scratch and persistent paths from veloc.cfg
+  // Since we know the paths from veloc.cfg, hard-code them here.
+  std::vector<std::string> dirs = {
+    "/tmp/openlb_veloc_scratch",
+    "/tmp/openlb_veloc_persistent"
+  };
+
+  for (const auto& dir : dirs) {
+    DIR* d = opendir(dir.c_str());
+    if (!d) continue;
+    struct dirent* ent;
+    while ((ent = readdir(d)) != nullptr) {
+      std::string fname(ent->d_name);
+      if (fname == "." || fname == "..") continue;
+      // Check if this file belongs to our checkpoint name
+      if (fname.rfind(name, 0) != 0) continue;
+      // Check if it contains the version we want to keep
+      // VeloC uses name-VVVVVV-RRRR patterns; if keep_version matches, skip
+      std::string ver_str = "-" + std::to_string(keep_version) + "-";
+      std::string ver_str2 = "-" + std::to_string(keep_version) + ".";
+      if (fname.find(ver_str) != std::string::npos ||
+          fname.find(ver_str2) != std::string::npos) {
+        continue; // Keep this file (it's the version we want)
+      }
+      // Also keep metadata/header files that don't have version numbers
+      // Check for numeric version in name before deleting
+      bool has_dash_digit = false;
+      for (size_t i = name.size(); i < fname.size(); ++i) {
+        if (fname[i] == '-' && i+1 < fname.size() && std::isdigit(fname[i+1])) {
+          has_dash_digit = true;
+          break;
+        }
+      }
+      if (!has_dash_digit) continue; // Skip non-versioned files
+      // Delete old version file
+      std::string full = dir + "/" + fname;
+      std::remove(full.c_str());
+    }
+    closedir(d);
+  }
+}
