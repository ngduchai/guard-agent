+static void writeField( FILE *fp, Field *f )
+{
+    if( !f || !f->data_ ) {
+        size_t zero = 0;
+        std::fwrite( &zero, sizeof(size_t), 1, fp );
+        return;
+    }
+    size_t count = static_cast<size_t>( f->number_of_points_ );
+    std::fwrite( &count, sizeof(size_t), 1, fp );
+    std::fwrite( f->data_, sizeof(double), count, fp );

+// Read a single EM Field array from an open FILE*. Resizes if needed.
+static bool readField( FILE *fp, Field *f )
+{
+    size_t count = 0;
+    if( std::fread( &count, sizeof(size_t), 1, fp ) != 1 ) return false;
+    if( count == 0 ) return true; // field not present, skip
+    if( !f || !f->data_ ) return false; // field expected but missing
+    if( static_cast<size_t>( f->number_of_points_ ) != count ) {
+        // Size mismatch - this is an error but we try to read anyway
+        return false;
+    }
+    if( std::fread( f->data_, sizeof(double), count, fp ) != count ) return false;
+    return true;
+}

+// Write checkpoint data for this MPI rank to a VeloC-routed file
+// Returns true on success
+static bool smileiVelocCheckpoint( VectorPatch &vecPatches,
+                                   unsigned int itime,
+                                   double time_prim,
+                                   double time_dual )
+{
+    // Get the file path from VeloC
+    char routed[VELOC_MAX_NAME];
+    if( VELOC_Route_file( "smilei_state", routed ) != VELOC_SUCCESS ) {
+        std::cerr << "[VeloC] Failed to get route file for checkpoint" << std::endl;
+        return false;
+    }
+
+    FILE *fp = std::fopen( routed, "wb" );
+    if( !fp ) {
+        std::cerr << "[VeloC] Failed to open checkpoint file: " << routed << std::endl;
+        return false;
+    }
+
+    // Write header: itime, time_prim, time_dual
+    std::fwrite( &itime,     sizeof(unsigned int), 1, fp );
+    std::fwrite( &time_prim, sizeof(double),        1, fp );
+    std::fwrite( &time_dual, sizeof(double),        1, fp );
+
+    // Write number of patches
+    size_t npatches = vecPatches.size();
+    std::fwrite( &npatches, sizeof(size_t), 1, fp );
+
+    for( unsigned int ipatch = 0; ipatch < vecPatches.size(); ++ipatch ) {
+        ElectroMagn *EM = vecPatches( ipatch )->EMfields;
+
+        // Write all primary EM fields
+        writeField( fp, EM ? EM->Ex_   : nullptr );
+        writeField( fp, EM ? EM->Ey_   : nullptr );
+        writeField( fp, EM ? EM->Ez_   : nullptr );
+        writeField( fp, EM ? EM->Bx_   : nullptr );
+        writeField( fp, EM ? EM->By_   : nullptr );
+        writeField( fp, EM ? EM->Bz_   : nullptr );
+        writeField( fp, EM ? EM->Bx_m  : nullptr );
+        writeField( fp, EM ? EM->By_m  : nullptr );
+        writeField( fp, EM ? EM->Bz_m  : nullptr );
+        writeField( fp, EM ? EM->Jx_   : nullptr );
+        writeField( fp, EM ? EM->Jy_   : nullptr );
+        writeField( fp, EM ? EM->Jz_   : nullptr );
+        writeField( fp, EM ? EM->rho_  : nullptr );
+
+        // Write number of species
+        size_t nspecies = vecPatches( ipatch )->vecSpecies.size();
+        std::fwrite( &nspecies, sizeof(size_t), 1, fp );
+
+        for( unsigned int ispec = 0; ispec < nspecies; ++ispec ) {
+            Particles *parts = vecPatches( ipatch )->vecSpecies[ispec]->particles;
+            if( !parts ) {
+                // No particles: write zero count for all arrays
+                size_t zero = 0;
+                // Position dimensions + data
+                size_t ndim = 0;
+                std::fwrite( &ndim, sizeof(size_t), 1, fp );
+                std::fwrite( &zero, sizeof(size_t), 1, fp ); // Momentum dims
+                std::fwrite( &zero, sizeof(size_t), 1, fp ); // npart
+                continue;
+            }
+
+            size_t npart  = static_cast<size_t>( parts->size() );
+            size_t nposdim = static_cast<size_t>( parts->Position.size() );
+            size_t nmomdim = static_cast<size_t>( parts->Momentum.size() );
+
+            std::fwrite( &npart,   sizeof(size_t), 1, fp );
+            std::fwrite( &nposdim, sizeof(size_t), 1, fp );
+            std::fwrite( &nmomdim, sizeof(size_t), 1, fp );
+
+            if( npart > 0 ) {
+                // Position
+                for( size_t d = 0; d < nposdim; ++d ) {
+                    std::fwrite( parts->Position[d].data(), sizeof(double), npart, fp );
+                }
+                // Momentum
+                for( size_t d = 0; d < nmomdim; ++d ) {
+                    std::fwrite( parts->Momentum[d].data(), sizeof(double), npart, fp );
+                }
+                // Weight
+                std::fwrite( parts->Weight.data(), sizeof(double), npart, fp );
+                // Charge
+                std::fwrite( parts->Charge.data(), sizeof(short), npart, fp );
+                // Chi (quantum parameter)
+                size_t has_chi = parts->Chi.size() > 0 ? 1 : 0;
+                std::fwrite( &has_chi, sizeof(size_t), 1, fp );
+                if( has_chi ) {
+                    std::fwrite( parts->Chi.data(), sizeof(double), npart, fp );
+                }
+                // Tau (Monte-Carlo optical depth)
+                size_t has_tau = parts->Tau.size() > 0 ? 1 : 0;
+                std::fwrite( &has_tau, sizeof(size_t), 1, fp );
+                if( has_tau ) {
+                    std::fwrite( parts->Tau.data(), sizeof(double), npart, fp );
+                }
+                // Id (tracking)
+                size_t has_id = parts->Id.size() > 0 ? 1 : 0;
+                std::fwrite( &has_id, sizeof(size_t), 1, fp );
+                if( has_id ) {
+                    std::fwrite( parts->Id.data(), sizeof(uint64_t), npart, fp );
+                }
+                // Position_old
+                size_t nold = static_cast<size_t>( parts->Position_old.size() );
+                std::fwrite( &nold, sizeof(size_t), 1, fp );
+                for( size_t d = 0; d < nold; ++d ) {
+                    std::fwrite( parts->Position_old[d].data(), sizeof(double), npart, fp );
+                }
+            }
+        }
+    }
+
+    std::fclose( fp );
+    return true;
+}
